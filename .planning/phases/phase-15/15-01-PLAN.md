      exitCode: number; // 0 = all pass, 2 = auth issue, 3 = validation issue, 1 = other

    Implement checks as separate functions:

    **checkNodeVersion(): DoctorCheck**
    - Parse `process.version` (e.g., "v20.11.0"), extract major version
    - Major >= 22: status 'pass', message "Node.js v{version} (>= 20)"
    - Major >= 20: status 'pass', message "Node.js v{version}"
    - Major < 20: status 'warn', message "Node.js v{version} (>= 20 recommended)"

    **checkNpm(): DoctorCheck**
    - Run `execSync('npm --version', { encoding: 'utf8', timeout: 5000, stdio: ['pipe', 'pipe', 'pipe'] })`
    - On success: status 'pass', message "npm v{version}"
    - On error: status 'fail', message "npm not found in PATH"

    **checkApiKey(): DoctorCheck**
    - Call `loadConfig(projectDir, { noExit: true })` -- if it throws, key is missing
    - On throw: status 'fail', message "JULES_API_KEY not set (set in .env, environment, or pass --api-key)"
    - On success: status 'pass', message "JULES_API_KEY is set"

    **checkApiConnectivity(config: JulesConfig): Promise<DoctorCheck>**
    - Create `new JulesClient(config)`, call `client.listSources()` (no pageToken, default pageSize is fine)
    - On success: status 'pass', message "Jules API reachable"
    - On error with status 401/403: status 'fail', message "Jules API auth failed -- check JULES_API_KEY"
    - On other error: status 'fail', message "Jules API unreachable: {error.message}"

    **checkTaskFile(filePath: string): DoctorCheck**
    - Check `existsSync(filePath)` -- fail if not: "File not found: {path}"
    - Call `loadTasks(filePath)` in try/catch
    - On success: status 'pass', message "Task file valid ({count} task(s))"
    - On error: status 'fail', message "Task file invalid: {error.message}"

    **runDoctor(projectDir: string, options?: { apiKeyOverride?: string; taskFile?: string }): Promise<DoctorResult>**
    - Run node_version, npm, api_key checks (sync)
    - If api_key passes, run api_connectivity (async)
    - If taskFile provided, run task_file check
    - Compute exitCode: all pass = 0, any auth fail = 2, any validation fail = 3, other fail = 1
    - Return { checks, exitCode }

    **Step 3: Wire CLI command**

    In `src/cli.ts`, add after the `cancel` command and before `tail`:

    ```typescript
    // ---------- doctor ----------
    program
      .command('doctor')
      .description('Validate environment: Node.js, npm, JULES_API_KEY, API connectivity, task file')
      .option('--task-file <path>', 'validate a task file (YAML or JSON)')
      .action(async (opts: { taskFile?: string }) => {
        const { runDoctor } = await import('./doctor.js');
        const opts_global = program.opts() as { project: string; apiKey?: string; json?: boolean };
        const projectDir = resolve(opts_global.project);

        const result = await runDoctor(projectDir, {
          apiKeyOverride: opts_global.apiKey,
          taskFile: opts.taskFile,
        });

        emit(
          () => {
            console.log(chalk.bold('\nDoctor Check\n'));
            for (const check of result.checks) {
              const icon = check.status === 'pass' ? chalk.green('✓')
                : check.status === 'warn' ? chalk.yellow('⚠')
                : chalk.red('✗');
              console.log(`  ${icon} ${check.name}: ${check.message}`);
            }
            console.log('');
          },
          { checks: result.checks },
        );

        if (result.exitCode !== 0) process.exit(result.exitCode);
      });
    ```
