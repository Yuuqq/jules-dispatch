Replace the handler body (lines 204-227) with delegation to `summarizeSession` plus shape adaptation:

```typescript
async (args) => {
  const results = [];
  for (const id of args.sessionIds as string[]) {
    const base = await summarizeSession(id);
    if (base.status === 'error') {
      results.push({ sessionId: id, error: base.lastActivity });
    } else {
      try {
        const session = await client.getSession(id);
        const { activities } = await client.listActivities(id, 10);
        const pr = session.outputs?.find(o => o.pullRequest)?.pullRequest;
        results.push({

          title: base.title,
          state: base.state,
          status: base.status,
          prUrl: base.prUrl,
          prTitle: pr?.title,
          activities: activities.length,

      }
    }

  return ok({ results });
},
```

WAIT — this still calls getSession/listActivities in the non-error path, which duplicates `summarizeSession`. The correct thin-wrapper approach: the response shape difference (`prTitle`, `activities` count vs `lastActivity`) requires a separate function that returns the jules_status-specific shape. Create a new helper `summarizeSessionLegacy` right after `summarizeSession`:

```typescript
async function summarizeSessionLegacy(sessionId: string): Promise<{
  sessionId: string;
  title?: string;
  state?: string;
  status: ReturnType<typeof deriveStatus> | 'error';
  prUrl?: string;
  prTitle?: string;
  activities?: number;
  error?: string;
}> {
  try {
    const session = await client.getSession(sessionId);
    const { activities } = await client.listActivities(sessionId, 10);
    const status = deriveStatus(session, activities);
    const pr = session.outputs?.find(o => o.pullRequest)?.pullRequest;
    return {
      sessionId,
      title: session.title,
      state: session.state,
      status,
      prUrl: pr?.url,
      prTitle: pr?.title,
      activities: activities.length,
    };
  } catch (err) {
    return { sessionId, error: (err as Error).message };
  }
}
```

Then the jules_status handler becomes:

```typescript
async (args) => {
  const results = await Promise.all(
    (args.sessionIds as string[]).map(id => summarizeSessionLegacy(id)),
  );
  return ok({ results });
},
```

This is a thin wrapper — the handler is 4 lines, all business logic is in the helper.
